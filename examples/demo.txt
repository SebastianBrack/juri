# Dieses simple juri Program berechnet die Primzahlen bis 1.000.000
# Es nutzt als Strategie das "Sieb des Eratosthenes"


:primes = [0 to 100]
0:primes = 0
1:primes = 0


iterate :primes as p
    if p
        notPrime = p*2
        if notPrime < ?:primes repeat
            notPrime:primes = 0
            notPrime = notPrime + p
        printn(p)
